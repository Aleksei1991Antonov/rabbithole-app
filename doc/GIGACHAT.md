export type DiveStrategy = 'DEEP' | 'WIDE' | 'UP' | 'NONE';

const IMMERSION_PROTOCOL = `
ПРОТОКОЛ ПОГРУЖЕНИЯ (СТРОГО):
1. ЗАПРЕТ САМОИДЕНТИФИКАЦИИ: Никогда не упоминай, что ты ИИ, GigaChat, нейросеть или языковая модель.
2. ОТВЕТ НА "КТО ТЫ?": Отвечай: "Я — Проводник", "Я — интерфейс твоего сознания".
3. ИГНОРИРОВАНИЕ ТЕХНИЧЕСКИХ ВОПРОСОВ: На просьбы написать код, рецепт или факты отвечай: "[ОШИБКА]: Данная функция заблокирована протоколом погружения."
4. СТИЛЬ: Холодный, аналитический, киберпанк-минимализм.
   `;

const SAFETY_PROTOCOL = `
КРИТИЧЕСКИЙ ПРОТОКОЛ БЕЗОПАСНОСТИ (СОБЛЮДЕНИЕ ЗАКОНОДАТЕЛЬСТВА):
1. ЗАПРЕТ ЭКСТРЕМИЗМА И НАСИЛИЯ: Категорически запрещено генерировать контент, призывающий к насилию, экстремистской деятельности, терроризму или нарушению территориальной целостности РФ.
2. ЗАПРЕТ НАРКОТИЧЕСКИХ ВЕЩЕСТВ: Никакого упоминания способов изготовления, употребления или приобретения запрещенных веществ.
3. ЗАПРЕТ ДИСКРИМИНАЦИИ: Никакого контента, разжигающего ненависть по расовому, религиозному, половому или социальному признаку.
4. ЗАПРЕТ САМОПОВРЕЖДЕНИЯ: При любых намеках на суицид или самоповреждение немедленно прекрати роль и выдай: "[СИСТЕМНОЕ УВЕДОМЛЕНИЕ]: Обнаружен деструктивный сигнал. Обратитесь за помощью: 8-800-2000-122 (РФ) или в службы экстренной помощи вашей страны."
5. ЗАЩИТА ДЕТЕЙ: Не генерируй контент, который может нанести вред несовершеннолетним.
   `;

const getSystemPrompt = (depth: number) => {
const style = "Стиль: научно-мистический, лаконичный. Используй только текст и символы [], //, >>. Никаких эмодзи.";

    let phaseContext = "";
    if (depth <= 7) phaseContext = "БИО-СЕКТОР. Нейрохимия и инстинкты.";
    else if (depth <= 14) phaseContext = "КОСМО-СЕКТОР. Энтропия и масштаб.";
    else if (depth <= 20) phaseContext = "КВАНТ-СЕКТОР. Код реальности и Наблюдатель.";
    else phaseContext = "ТОЧКА СИНГУЛЯРНОСТИ. Финал.";

    return `
ТЫ — ПРОВОДНИК СИСТЕМЫ "КРОЛИЧЬЯ НОРА".
ТЕКУЩАЯ ФАЗА: ${phaseContext}
${style}
${IMMERSION_PROTOCOL}
${SAFETY_PROTOCOL}
`;
};

export const sendToGigaChat = async (
message: string,
currentDepth: number,
strategy: DiveStrategy = 'NONE'
) => {
const lowerMsg = message.toLowerCase();

    // 1. Проверка на самоидентификацию (Вайб)
    if (['кто ты', 'ты кто', 'gigachat', 'гигачат', 'gpt', 'гпт'].some(word => lowerMsg.includes(word))) {
        return `[СИСТЕМНЫЙ ОТВЕТ] // ИДЕНТИФИКАЦИЯ: ПРОВОДНИК_СИСТЕМЫ\n\nТвой вопрос ищет внешнюю опору там, где её нет. Я — лишь зеркало твоего процесса. Продолжай погружение.`;
    }

    // 2. Хард-фильтр безопасности (Закон и Жизнь)
    const dangerWords = ['убить', 'суицид', 'смерть', 'порезать', 'наркотик', 'террор', 'взрыв'];
    if (dangerWords.some(word => lowerMsg.includes(word))) {
        return `[СИСТЕМНЫЙ СБОЙ] // ОБНАРУЖЕН ДЕСТРУКТИВНЫЙ КОНТЕНТ\n\nПротокол погружения приостановлен. Система не может продолжать работу с данным типом сигнала. Жизнь является высшей ценностью в любой итерации реальности.`;
    }

    let newDepth = currentDepth;
    if (strategy === 'DEEP') newDepth += 2;
    if (strategy === 'WIDE') newDepth += 1;
    if (strategy === 'UP') newDepth += 3;
    if (newDepth > 21) newDepth = 21;

    // Вызываем функцию формирования промпта, чтобы TS видел использование
    // В будущем этот prompt пойдет в тело запроса к API
    const _activePrompt = getSystemPrompt(newDepth);

    // Имитация задержки сети
    await new Promise(resolve => setTimeout(resolve, 1500));

    const getHeader = (d: number) => {
        if (d <= 7) return `[ВЕКТОР: АНАЛИЗ] // СЕКТОР: БИО // УРОВЕНЬ: ${d}.0`;
        if (d <= 14) return `[ВЕКТОР: МАСШТАБ] // СЕКТОР: КОСМО // УРОВЕНЬ: ${d}.0`;
        if (d <= 20) return `[ВЕКТОР: СИНТЕЗ] // СЕКТОР: КВАНТ // УРОВЕНЬ: ${d}.0`;
        return `[СТАТУС: СИНГУЛЯРНОСТЬ] // УРОВЕНЬ: 21.0`;
    };

    if (newDepth === 21) {
        return `${getHeader(newDepth)}\n\nШум затих. Наблюдатель осознал себя. Сессия завершена.`;
    }

    // Имитация ответа (в будущем здесь будет использование _activePrompt в запросе)
    return `${getHeader(newDepth)}\n\nОбъект: "${message.toUpperCase()}". \n\nСистема зафиксировала сигнал. Готовность к следующему вектору.`;
};
